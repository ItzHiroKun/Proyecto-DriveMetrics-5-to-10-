-- Base de datos: facturacion_db
CREATE TABLE facturas (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    numero_boleta BIGINT NOT NULL UNIQUE,
    fecha_emision DATETIME DEFAULT CURRENT_TIMESTAMP,
    orden_id BIGINT NOT NULL UNIQUE,
    subtotal_mano_obra DECIMAL(12,2) DEFAULT 0,
    subtotal_repuestos DECIMAL(12,2) DEFAULT 0,
    iva DECIMAL(12,2) NOT NULL,
    total_final DECIMAL(12,2) NOT NULL
);

CREATE TABLE pagos (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    factura_id BIGINT NOT NULL,
    monto DECIMAL(12,2) NOT NULL,
    fecha_pago DATETIME DEFAULT CURRENT_TIMESTAMP,
    metodo_pago VARCHAR(50),
    referencia VARCHAR(100),
    FOREIGN KEY (factura_id) REFERENCES facturas(id)
);
